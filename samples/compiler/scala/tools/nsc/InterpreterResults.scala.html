package scala.tools.nsc

object object scala.tools.nsc.InterpreterResultsInterpreterResults {

  /** A result from interpreting one line of input. */
  abstract sealed class class Result extends java.lang.Object with ScalaObjectResult

  /** The line was interpreted successfully. */
  case object object scala.tools.nsc.InterpreterResults.SuccessSuccess extends scala.tools.nsc.InterpreterResults.ResultResult

  /** The line was erroneous in some way. */
  case object object scala.tools.nsc.InterpreterResults.ErrorError extends scala.tools.nsc.InterpreterResults.ResultResult
  
  /** The input was incomplete.  The caller should request more
   *  input.
   */
  case object object scala.tools.nsc.InterpreterResults.IncompleteIncomplete extends scala.tools.nsc.InterpreterResults.ResultResult

}
