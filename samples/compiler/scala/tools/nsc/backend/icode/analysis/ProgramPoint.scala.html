package scala.tools.nsc.backend.icode.analysis

/** Program points are locations in the program where we want to
 *  assert certain properties through data flow analysis, e.g.
 *  basic blocks.
 */
trait trait ProgramPoint[a <: scala.tools.nsc.backend.icode.analysis.ProgramPoint[a]] extends java.lang.ObjectProgramPoint[>: Nothing <: scala.tools.nsc.backend.icode.analysis.ProgramPoint[a]a <: ProgramPoint[a]] {
  def => List[a]predecessors: List[a]List[a]
  def => List[a]successors: List[a]List[a]
}
