package scalax.collection

/** Variant of <code>Iterable</code> which also demands 
 *  implementation of a `size` method. 
 *  Basically, this trait just adds size to Iterable,
 *  and provides an optimized implementation of toArray based on it.
 *
 *  @author Martin Odersky
 *  @version 2.8
 */
trait trait SizedIterable[+A] extends java.lang.Object with scalax.collection.Iterable[A] with ScalaObjectSizedIterable[+>: Nothing <: AnyA] extends scalax.collection.Iterable[A]Iterable[A] {

  /** Returns the number of elements in this collection.
    *
    *  @return number of collection elements.
    */
  def => Intsize : IntInt

  /** Converts this iterable to a fresh Array with <code>size</code> elements.
    */
  override def [B >: A]Array[B]toArray[>: A <: AnyB >: A]: Array[B]Array[B] = {
    val Array[B]result = Array[B]new Array[B]Array[B](=> Intsize)
    (Array[B],Int)UnitcopyToArray(Array[B]result, Int(0)0)
    Array[B]result
  }
}
