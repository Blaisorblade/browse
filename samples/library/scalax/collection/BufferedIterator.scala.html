package scalax.collection

/** Buffered iterators are iterators which allow to inspect the next
 *  element without discarding it.
 *
 *  @author  Martin Odersky
 *  @version 2.8
 */
trait trait BufferedIterator[+A] extends java.lang.Object with scalax.collection.Iterator[A] with ScalaObjectBufferedIterator[+>: Nothing <: AnyA] extends scalax.collection.Iterator[A]Iterator[A] {

  /** Returns current element of iterator without advancing beyond it.
   */
  def => Ahead: AA

  override def => BufferedIterator.this.typebuffered: this.type = BufferedIterator.this.typethis
}
