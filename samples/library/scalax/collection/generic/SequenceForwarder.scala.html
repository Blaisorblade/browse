package scalax.collection.generic

/** This class implements a forwarder for sequences. It forwards
 *  all calls to a different sequence object except for
 *
 *    - toString, hashCode, equals, stringPrefix
 *    - newBuilder, view, toSequence
 *    - all calls creating a new iterable objetc of the same kind
 *
 *  The above methods are forwarded by subclass SequenceProxy
 *
 *  @author  Martin Odersky
 *  @version 2.8
 */
trait trait SequenceForwarder[+A] extends java.lang.Object with scalax.collection.Sequence[A] with scalax.collection.generic.IterableForwarder[A] with ScalaObjectSequenceForwarder[+>: Nothing <: AnyA] extends scalax.collection.Sequence[A]Sequence[A] with scalax.collection.generic.IterableForwarder[A]IterableForwarder[A] {

  protected override def => scalax.collection.Sequence[A]underlying: scalax.collection.Sequence[A]Sequence[A] 
  
  // PartialFunction delegates

  override def (Int)Aapply(Inti: IntInt): AA = => scalax.collection.Sequence[A]underlying.(Int)Aapply(Inti)
  override def (Int)BooleanisDefinedAt(Intx: IntInt): BooleanBoolean = => scalax.collection.Sequence[A]underlying.(Int)BooleanisDefinedAt(Intx)
  
  // Sequence delegates
  // Sequence methods could be printed by  cat SequenceTemplate.scala | sed -n '/trait Seq/,$ p' | egrep '^  (override )?def'
  
  override def => Intlength: IntInt = => scalax.collection.Sequence[A]underlying.=> Intlength
  override def (Int)IntlengthCompare(Intl: IntInt) = => scalax.collection.Sequence[A]underlying (Int)IntlengthCompare Intl
  override def ((A) => Boolean,Int)IntsegmentLength((A) => Booleanp: A => Boolean, Intfrom: IntInt): IntInt = => scalax.collection.Sequence[A]underlying.((A) => Boolean,Int)IntsegmentLength((A) => Booleanp, Intfrom)
  override def ((A) => Boolean)IntprefixLength((A) => Booleanp: A => Boolean) = => scalax.collection.Sequence[A]underlying.((A) => Boolean)IntprefixLength((A) => Booleanp)
  override def ((A) => Boolean,Int)IntindexWhere((A) => Booleanp: A => Boolean, Intfrom: IntInt): IntInt = => scalax.collection.Sequence[A]underlying.((A) => Boolean,Int)IntindexWhere((A) => Booleanp, Intfrom)
  override def [B >: A](B,Int)IntindexOf[>: A <: AnyB >: A](Belem: BB, Intfrom: IntInt): IntInt = => scalax.collection.Sequence[A]underlying.(B,Int)IntindexOf(Belem, Intfrom)
  override def => scalax.collection.Iterator[A]reversedElements: scalax.collection.Iterator[A]Iterator[A] = => scalax.collection.Sequence[A]underlying.=> scalax.collection.Iterator[A]reversedElements
  override def [B](scalax.collection.Sequence[B],Int)BooleanstartsWith[>: Nothing <: AnyB](scalax.collection.Sequence[B]that: scalax.collection.Sequence[B]Sequence[B], Intoffset: IntInt): BooleanBoolean = => scalax.collection.Sequence[A]underlying.(scalax.collection.Sequence[B],Int)BooleanstartsWith(scalax.collection.Sequence[B]that, Intoffset)
  override def [B](scalax.collection.Sequence[B])BooleanendsWith[>: Nothing <: AnyB](scalax.collection.Sequence[B]that: scalax.collection.Sequence[B]Sequence[B]): BooleanBoolean = => scalax.collection.Sequence[A]underlying.(scalax.collection.Sequence[B])BooleanendsWith(scalax.collection.Sequence[B]that)
  override def [B >: A](scalax.collection.Sequence[B])IntindexOf[>: A <: AnyB >: A](scalax.collection.Sequence[B]that: scalax.collection.Sequence[B]Sequence[B]): IntInt = => scalax.collection.Sequence[A]underlying.(scalax.collection.Sequence[B])IntindexOf(scalax.collection.Sequence[B]that)
  override def (Any)Booleancontains(Anyelem: AnyAny): BooleanBoolean = => scalax.collection.Sequence[A]underlying.(Any)Booleancontains(Anyelem)
  override def => Rangeindices: RangeRange = => scalax.collection.Sequence[A]underlying.=> Rangeindices
}
