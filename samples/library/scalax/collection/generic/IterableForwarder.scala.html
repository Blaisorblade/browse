package scalax.collection.generic

import collection.mutable.Buffer
import collection.immutable.{List, Stream}

/** This trait implements a forwarder for iterable objects. It forwards
 *  all calls to a different iterable object, except for
 *
 *    - toString, hashCode, equals, stringPrefix
 *    - newBuilder, view
 *    - all calls creating a new iterable objetc of the same kind
 *
 *  The above methods are forwarded by subclass IterableProxy
 *
 *  @author  Martin Odersky
 *  @version 2.8
 */
trait trait IterableForwarder[+A] extends java.lang.Object with scalax.collection.Iterable[A] with ScalaObjectIterableForwarder[+>: Nothing <: AnyA] extends scalax.collection.Iterable[A]Iterable[A] {

  /** The iterable object to which calls are forwarded */
  protected def => scalax.collection.Iterable[A]underlying: scalax.collection.Iterable[A]Iterable[A]
  
  // Iterable delegates
  // Iterable methods could be printed by  cat IterableTemplate.scala | sed -n '/trait Iterable/,$ p' | egrep '^  (override )?def'
  
  override def => scalax.collection.Iterator[A]elements = => scalax.collection.Iterable[A]underlying.=> scalax.collection.Iterator[A]elements
  override def => BooleanisEmpty = => scalax.collection.Iterable[A]underlying.=> BooleanisEmpty
  override def => BooleanhasDefiniteSize = => scalax.collection.Iterable[A]underlying.=> BooleanhasDefiniteSize
  override def ((A) => Unit)Unitforeach((A) => Unitf: A => Unit) = => scalax.collection.Iterable[A]underlying.((A) => Unit)Unitforeach((A) => Unitf)
  override def ((A) => Boolean)Booleanforall((A) => Booleanp: A => Boolean): BooleanBoolean = => scalax.collection.Iterable[A]underlying.((A) => Boolean)Booleanforall((A) => Booleanp)
  override def ((A) => Boolean)Booleanexists((A) => Booleanp: A => Boolean): BooleanBoolean = => scalax.collection.Iterable[A]underlying.((A) => Boolean)Booleanexists((A) => Booleanp)
  override def ((A) => Boolean)Intcount((A) => Booleanp: A => Boolean): IntInt = => scalax.collection.Iterable[A]underlying.((A) => Boolean)Intcount((A) => Booleanp)
  override def ((A) => Boolean)Option[A]find((A) => Booleanp: A => Boolean): Option[A]Option[A] = => scalax.collection.Iterable[A]underlying.((A) => Boolean)Option[A]find((A) => Booleanp)
  override def [B](B)((B, A) => B)BfoldLeft[>: Nothing <: AnyB](Bz: BB)((B, A) => Bop: (B, A) => B): BB = => scalax.collection.Iterable[A]underlying.(B)((B, A) => B)BfoldLeft(Bz)((B, A) => Bop)
  override def [B](B)((A, B) => B)BfoldRight[>: Nothing <: AnyB](Bz: BB)((A, B) => Bop: (A, B) => B): BB = => scalax.collection.Iterable[A]underlying.(B)((A, B) => B)BfoldRight(Bz)((A, B) => Bop)
  override def [B >: A]((B, A) => B)BreduceLeft[>: A <: AnyB >: A]((B, A) => Bop: (B, A) => B): BB = => scalax.collection.Iterable[A]underlying.((B, A) => B)BreduceLeft((B, A) => Bop)
  override def [B >: A]((A, B) => B)BreduceRight[>: A <: AnyB >: A]((A, B) => Bop: (A, B) => B): BB = => scalax.collection.Iterable[A]underlying.((A, B) => B)BreduceRight((A, B) => Bop)
  override def [B >: A](scalax.collection.mutable.Buffer[B])UnitcopyToBuffer[>: A <: AnyB >: A](scalax.collection.mutable.Buffer[B]dest: scalax.collection.mutable.Buffer[B]Buffer[B]) = => scalax.collection.Iterable[A]underlying.(scalax.collection.mutable.Buffer[B])UnitcopyToBuffer(scalax.collection.mutable.Buffer[B]dest)
  override def [B >: A](Array[B],Int,Int)UnitcopyToArray[>: A <: AnyB >: A](Array[B]xs: Array[B]Array[B], Intstart: IntInt, Intlen: IntInt) = => scalax.collection.Iterable[A]underlying.(Array[B],Int,Int)UnitcopyToArray(Array[B]xs, Intstart, Intlen)
  override def [B >: A]Array[B]toArray[>: A <: AnyB >: A]: Array[B]Array[B] = => scalax.collection.Iterable[A]underlying.Array[B]toArray
  override def => scalax.collection.immutable.List[A]toList: scalax.collection.immutable.List[A]List[A] = => scalax.collection.Iterable[A]underlying.=> scalax.collection.immutable.List[A]toList
  override def => scalax.collection.Sequence[A]toSequence: scalax.collection.Sequence[A]Sequence[A] = => scalax.collection.Iterable[A]underlying.=> scalax.collection.Sequence[A]toSequence
  override def => scalax.collection.immutable.Stream[A]toStream: scalax.collection.immutable.Stream[A]Stream[A] = => scalax.collection.Iterable[A]underlying.=> scalax.collection.immutable.Stream[A]toStream
  override def (String,String,String)StringmkString(Stringstart: StringString, Stringsep: StringString, Stringend: StringString): StringString = => scalax.collection.Iterable[A]underlying.(String,String,String)StringmkString(Stringstart, Stringsep, Stringend)
  override def (StringBuilder,String,String,String)StringBuilderaddString(StringBuilderb: StringBuilderStringBuilder, Stringstart: StringString, Stringsep: StringString, Stringend: StringString): StringBuilderStringBuilder = => scalax.collection.Iterable[A]underlying.(StringBuilder,String,String,String)StringBuilderaddString(StringBuilderb, Stringstart, Stringsep, Stringend)
  
  override def => Ahead: AA = => scalax.collection.Iterable[A]underlying.=> Ahead
}
