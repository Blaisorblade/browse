package scala.collection


/** This is a simple wrapper class for <a href="Set.html"
 *  target="contentFrame"><code>scala.collection.Set</code></a>.
 *  It is most useful for assembling customized set abstractions
 *  dynamically using object composition and forwarding.
 *
 *  @author  Matthias Zenger
 *  @author  Martin Odersky
 *  @version 2.0, 01/01/2007
 */
trait trait SetProxy[A] extends java.lang.Object with scala.collection.Set[A] with IterableProxy[A] with ScalaObjectSetProxy[>: Nothing <: AnyA] extends scala.collection.Set[A]Set[A] with IterableProxy[A]IterableProxy[A] {
  def => scala.collection.Set[A]self: scala.collection.Set[A]Set[A]
  def => Intsize: IntInt = => scala.collection.Set[A]self.=> Intsize
  override def => BooleanisEmpty: BooleanBoolean = => scala.collection.Set[A]self.=> BooleanisEmpty
  def (A)Booleancontains(Aelem: AA): BooleanBoolean = => scala.collection.Set[A]self.(A)Booleancontains(Aelem)
  override def (scala.collection.Set[A])BooleansubsetOf(scala.collection.Set[A]that: scala.collection.Set[A]Set[A]): BooleanBoolean = => scala.collection.Set[A]self.(scala.collection.Set[A])BooleansubsetOf(scala.collection.Set[A]that)
}
