package scala.collection.mutable


/** This class should be used as a mixin. It synchronizes the <code>Map</code>
 *  functions of the class into which it is mixed in.
 *
 *  @author  Matthias Zenger, Martin Odersky
 *  @version 2.0, 31/12/2006
 */
trait trait SynchronizedMap[A, B] extends java.lang.Object with scala.collection.mutable.Map[A,B] with ScalaObjectSynchronizedMap[>: Nothing <: AnyA, >: Nothing <: AnyB] extends scala.collection.mutable.Map[A,B]Map[A, B] {

  abstract override def => Intsize: IntInt = (Int)Intsynchronized {
    => Intsuper.=> Intsize
  }

  abstract override def (A)Option[B]get(Akey: AA): Option[B]Option[B] = (Option[B])Option[B]synchronized {
    (A)Option[B]super.(A)Option[B]get(Akey)
  }

  override def => BooleanisEmpty: BooleanBoolean = (Boolean)Booleansynchronized {
    => Booleansuper.=> BooleanisEmpty
  }

  override def (A)Bapply(Akey: AA): BB = (B)Bsynchronized {
    (A)Bsuper.(A)Bapply(Akey)
  }

  override def (A)Booleancontains(Akey: AA): BooleanBoolean = (Boolean)Booleansynchronized {
    (A)Booleansuper.(A)Booleancontains(Akey)
  }

  override def (A)BooleanisDefinedAt(Akey: AA) = (Boolean)Booleansynchronized {
    (A)Booleansuper.(A)BooleanisDefinedAt(Akey)
  }

  override def => Iterator[A]keys: Iterator[A]Iterator[A] = (Iterator[A])Iterator[A]synchronized {
    => Iterator[A]super.=> Iterator[A]keys
  }

  override def => scala.collection.Set[A]keySet: collection.scala.collection.Set[A]Set[A] = (scala.collection.Set[A])scala.collection.Set[A]synchronized {
    => scala.collection.Set[A]super.=> scala.collection.Set[A]keySet
  }

  override def => Iterator[B]values: Iterator[B]Iterator[B] = (Iterator[B])Iterator[B]synchronized {
    => Iterator[B]super.=> Iterator[B]values
  }

  abstract override def => Iterator[(A, B)]elements: Iterator[(A, B)]Iterator[(A, B)] = (Iterator[(A, B)])Iterator[(A, B)]synchronized {
    => Iterator[(A, B)]super.=> Iterator[(A, B)]elements
  }

  override def => List[(A, B)]toList: List[(A, B)]List[(A, B)] = (List[(A, B)])List[(A, B)]synchronized {
    => List[(A, B)]super.=> List[(A, B)]toList

  abstract override def (A,B)Unitupdate(Akey: AA, Bvalue: BB): UnitUnit = (Unit)Unitsynchronized {
    (A,B)Unitsuper.(A,B)Unitupdate(Akey, Bvalue)

  override def ((A, B))Unit+= ((A, B)kv: (A, B)(A, B)): UnitUnit = (Unit)Unitsynchronized {

  }

   *  @param    kv2 the second key/first value pair.
   *  @param    kvs the remaining key/first value pairs.
   */
  override def ((A, B),(A, B),(A, B)*)Unit+= ((A, B)kv1: (A, B)(A, B), (A, B)kv2: (A, B)(A, B), (A, B)*kvs: (A, B)*(A, B)*): UnitUnit = (Unit)Unitsynchronized {
    ((A, B),(A, B),(A, B)*)Unitsuper.((A, B),(A, B),(A, B)*)Unit+=((A, B)kv1, (A, B)kv2, (A, B)*kvs: _*)
  }

  override def (Iterable[(A, B)])Unit++=(Iterable[(A, B)]map: Iterable[(A, B)]Iterable[(A, B)]): UnitUnit = (Unit)Unitsynchronized {
    (Iterable[(A, B)])Unitsuper.(Iterable[(A, B)])Unit++=(Iterable[(A, B)]map)

  override def (Iterator[(A, B)])Unit++=(Iterator[(A, B)]it: Iterator[(A, B)]Iterator[(A, B)]): UnitUnit = (Unit)Unitsynchronized {

  @deprecateddeprecated
  override def ((A, B)*)Unitincl((A, B)*mappings: (A, B)*(A, B)*): UnitUnit = (Unit)Unitsynchronized {

  }

  abstract override def (A)Unit-=(Akey: AA): UnitUnit = (Unit)Unitsynchronized {

  override def (A,A,A*)Unit-= (Akey1: AA, Akey2: AA, A*keys: A*A*): UnitUnit = (Unit)Unitsynchronized {

  override def (Iterable[A])Unit--=(Iterable[A]keys: Iterable[A]Iterable[A]): UnitUnit = (Unit)Unitsynchronized {

  @deprecateddeprecated
  override def (A*)Unitexcl(A*keys: A*A*): UnitUnit = (Unit)Unitsynchronized {
    (A*)Unitsuper.(A*)Unitexcl(A*keys: _*)

  override def ()Unitclear(): UnitUnit = (Unit)Unitsynchronized {

  }

  override def ()StringtoString() = (String)Stringsynchronized {
    ()Stringsuper.()StringtoString()
  }

  override def (Any)Booleanequals(Anythat: AnyAny): BooleanBoolean = (Boolean)Booleansynchronized {
    (Any)Booleansuper.(Any)Booleanequals(Anythat)
  }
  
  override def ()InthashCode(): IntInt = (Int)Intsynchronized {
    ()Intsuper.()InthashCode()
  }

  override def (scala.collection.mutable.Message[(A, B)])Unit<<(scala.collection.mutable.Message[(A, B)]cmd: scala.collection.mutable.Message[(A, B)]Message[(A, B)]): UnitUnit = (Unit)Unitsynchronized {
    (scala.collection.mutable.Message[(A, B)])Unitsuper.(scala.collection.mutable.Message[(A, B)])Unit<<(scala.collection.mutable.Message[(A, B)]cmd)
  }

  override def ()scala.collection.mutable.Map[A,B]clone(): scala.collection.mutable.Map[A,B]Map[A, B] = (scala.collection.mutable.Map[A,B])scala.collection.mutable.Map[A,B]synchronized {
    ()scala.collection.mutable.Map[A,B]super.()scala.collection.mutable.Map[A,B]clone()
  }
}
