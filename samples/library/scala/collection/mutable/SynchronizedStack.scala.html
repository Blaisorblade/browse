// $Id: SynchronizedStack.scala 16893 2009-01-13 13:09:22Z cunei $


package scala.collection.mutable


/** This is a synchronized version of the <code>Stack[T]</code> class. It
 *  implements a data structure which allows to store and retrieve
 *  objects in a last-in-first-out (LIFO) fashion.
 * 
 *  @author  Matthias Zenger
 *  @version 1.0, 03/05/2004
 */
class class SynchronizedStack[A] extends scala.collection.mutable.Stack[A] with ScalaObjectSynchronizedStack[>: Nothing <: AnyA] extends scala.collection.mutable.Stack[A]Stack[A] {

  /** Checks if the stack is empty.
   *
   *  @return true, iff there is no element on the stack
   */
  override def => BooleanisEmpty: BooleanBoolean = (Boolean)Booleansynchronized { super.=> BooleanisEmpty }

  /** Pushes a single element on top of the stack.
   *
   *  @param  elem        the element to push onto the stack
   */
  override def (A)Unit+=(Aelem: AA): UnitUnit = (Unit)Unitsynchronized { super.(A)Unit+=(Aelem) }

  /** Pushes all elements provided by an <code>Iterable</code> object
   *  on top of the stack. The elements are pushed in the order they
   *  are given out by the iterator.
   *
   *  @param  iter        an iterable object
   */
  override def (Iterable[A])Unit++=(Iterable[A]iter: Iterable[A]Iterable[A]): UnitUnit = (Unit)Unitsynchronized { super.(Iterable[A])Unit++=(Iterable[A]iter) }

  /** Pushes all elements provided by an iterator
   *  on top of the stack. The elements are pushed in the order they
   *  are given out by the iterator.
   *
   *  @param  iter        an iterator
   */
    override def (Iterator[A])Unit++=(Iterator[A]it: Iterator[A]Iterator[A]): UnitUnit = (Unit)Unitsynchronized { super.(Iterator[A])Unit++=(Iterator[A]it) }

  /** Pushes a sequence of elements on top of the stack. The first element
   *  is pushed first, etc.
   *
   *  @param  elems       a sequence of elements
   */
  override def (A*)Unitpush(A*elems: A*A*): UnitUnit = (Unit)Unitsynchronized { super.(Iterable[A])Unit++=(A*elems) }

  /** Returns the top element of the stack. This method will not remove
   *  the element from the stack. An error is signaled if there is no
   *  element on the stack.
   *
   *  @return the top element
   */
  override def => Atop: AA = (A)Asynchronized { super.=> Atop }

  /** Removes the top element from the stack.
   */
  override def ()Apop(): AA = (A)Asynchronized { super.()Apop }

  /**
   * Removes all elements from the stack. After this operation completed,
   * the stack will be empty.
   */
  override def ()Unitclear(): UnitUnit = (Unit)Unitsynchronized { super.()Unitclear }

  /** Returns an iterator over all elements on the stack. This iterator
   *  is stable with respect to state changes in the stack object; i.e.
   *  such changes will not be reflected in the iterator. The iterator
   *  issues elements in the order they were inserted into the stack
   *  (FIFO order).
   *
   *  @return an iterator over all stack elements.
   */
  override def => Iterator[A]elements: Iterator[A]Iterator[A] = (Iterator[A])Iterator[A]synchronized { super.=> Iterator[A]elements }

  /** Creates a list of all stack elements in FIFO order.
   *
   *  @return the created list.
   */
  override def => List[A]toList: List[A]List[A] = (List[A])List[A]synchronized { super.=> List[A]toList }

  /** Checks if two stacks are structurally identical.
   *
   *  @return true, iff both stacks contain the same sequence of elements.
   */
    override def (Any)Booleanequals(Anythat: AnyAny): BooleanBoolean = (Boolean)Booleansynchronized { super.(Any)Booleanequals(Anythat) }

  /** The hashCode method always yields an error, since it is not
   *  safe to use mutable stacks as keys in hash tables.
   *
   *  @return never.
   */
  override def ()InthashCode(): IntInt = (Int)Intsynchronized { super.()InthashCode() }

  /** Returns a textual representation of a stack as a string.
   *
   *  @return the string representation of this stack.
   */
  override def ()StringtoString = (String)Stringsynchronized { super.()StringtoString }
}
