package scala.collection.mutable

/** The J2ME version of the library defined this trait with a clone method
 * to substitute for the lack of Object.clone there
 */
trait trait CloneableCollection extends java.lang.Object with ScalaObjectCloneableCollection {
  override def ()AnyRefclone(): AnyRefAnyRef = ()java.lang.Objectsuper.()java.lang.Objectclone()
}
