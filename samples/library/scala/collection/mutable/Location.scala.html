package scala.collection.mutable


/** Class <code>Location</code> describes locations in messages implemented
 *  by class <code>Message</code>.
 *
 *  @author  Matthias Zenger
 *  @version 1.0, 10/05/2004
 */
abstract class class Location extends java.lang.Object with ScalaObjectLocation

case object object scala.collection.mutable.NANA extends scala.collection.mutable.LocationLocation

case object object scala.collection.mutable.StartStart extends scala.collection.mutable.LocationLocation

case object object scala.collection.mutable.EndEnd extends scala.collection.mutable.LocationLocation

case class class Index extends scala.collection.mutable.Location with ScalaObject with ProductIndex(Intn: IntInt) extends scala.collection.mutable.LocationLocation
