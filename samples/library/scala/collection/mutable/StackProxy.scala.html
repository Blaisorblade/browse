package scala.collection.mutable


/** A stack implements a data structure which allows to store and retrieve
 *  objects in a last-in-first-out (LIFO) fashion.
 *
 *  @author  Matthias Zenger
 *  @version 1.0, 10/05/2004
 */
trait trait StackProxy[A] extends scala.collection.mutable.Stack[A] with SeqProxy[A] with ScalaObjectStackProxy[>: Nothing <: AnyA] extends scala.collection.mutable.Stack[A]Stack[A] with SeqProxy[A]SeqProxy[A] {

  def => scala.collection.mutable.Stack[A]self: scala.collection.mutable.Stack[A]Stack[A]

  /** Access element number <code>n</code>.
   *
   *  @return  the element at index <code>n</code>.
   */
  override def (Int)Aapply(Intn: IntInt): AA = => scala.collection.mutable.Stack[A]self.(Int)Aapply(Intn)

  /** Returns the length of this stack.
   */
  override def => Intlength: IntInt = => scala.collection.mutable.Stack[A]self.=> Intlength

  /** Checks if the stack is empty.
   *
   *  @return true, iff there is no element on the stack
   */
  override def => BooleanisEmpty: BooleanBoolean = => scala.collection.mutable.Stack[A]self.=> BooleanisEmpty

  /** Pushes a single element on top of the stack.
   *
   *  @param  elem        the element to push onto the stack
   */
  override def (A)Unit+=(Aelem: AA): UnitUnit = => scala.collection.mutable.Stack[A]self (A)Unit+= Aelem

  /** Pushes all elements provided by an <code>Iterable</code> object
   *  on top of the stack. The elements are pushed in the order they
   *  are given out by the iterator.
   *
   *  @param  iter        an iterable object
   */
    override def (Iterable[A])Unit++=(Iterable[A]iter: Iterable[A]Iterable[A]): UnitUnit = => scala.collection.mutable.Stack[A]self (Iterable[A])Unit++= Iterable[A]iter

  /** Pushes all elements provided by an iterator
   *  on top of the stack. The elements are pushed in the order they
   *  are given out by the iterator.
   *
   *  @param  iter        an iterator
   */
  override def (Iterator[A])Unit++=(Iterator[A]it: Iterator[A]Iterator[A]): UnitUnit = => scala.collection.mutable.Stack[A]self (Iterator[A])Unit++= Iterator[A]it

  /** Pushes a sequence of elements on top of the stack. The first element
   *  is pushed first, etc.
   *
   *  @param  elems       a sequence of elements
   */
  override def (A*)Unitpush(A*elems: A*A*): UnitUnit = => scala.collection.mutable.Stack[A]self (Iterable[A])Unit++= A*elems

  /** Returns the top element of the stack. This method will not remove
   *  the element from the stack. An error is signaled if there is no
   *  element on the stack.
   *
   *  @return the top element
   */
  override def => Atop: AA = => scala.collection.mutable.Stack[A]self.=> Atop

  /** Removes the top element from the stack.
   */
  override def ()Apop(): AA = => scala.collection.mutable.Stack[A]self.()Apop

  /**
   * Removes all elements from the stack. After this operation completed,
   * the stack will be empty.
   */
  override def ()Unitclear(): UnitUnit = => scala.collection.mutable.Stack[A]self.()Unitclear

  /** Returns an iterator over all elements on the stack. This iterator
   *  is stable with respect to state changes in the stack object; i.e.
   *  such changes will not be reflected in the iterator. The iterator
   *  issues elements in the order they were inserted into the stack
   *  (FIFO order).
   *
   *  @return an iterator over all stack elements.
   */
  override def => Iterator[A]elements: Iterator[A]Iterator[A] = => scala.collection.mutable.Stack[A]self.=> Iterator[A]elements

  /** Creates a list of all stack elements in FIFO order.
   *
   *  @return the created list.
   */
  override def => List[A]toList: List[A]List[A] = => scala.collection.mutable.Stack[A]self.=> List[A]toList

  /** This method clones the stack.
   *
   *  @return  a stack with the same elements.
   */
  override def ()scala.collection.mutable.Stack[A]clone(): scala.collection.mutable.Stack[A]Stack[A] = template $anon extends scala.collection.mutable.Stack[A] with scala.collection.mutable.StackProxy[A]new scala.collection.mutable.StackProxy[A]StackProxy[A] {
    def => scala.collection.mutable.Stack[A]self = StackProxy.this.typeStackProxy.this.=> scala.collection.mutable.Stack[A]self.()scala.collection.mutable.Stack[A]clone()
  }
}
