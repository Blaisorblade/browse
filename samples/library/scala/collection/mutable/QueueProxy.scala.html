package scala.collection.mutable


/** <code>Queue</code> objects implement data structures that allow to
 *  insert and retrieve elements in a first-in-first-out (FIFO) manner.
 *
 *  @author  Matthias Zenger
 *  @version 1.1, 03/05/2004
 */
trait trait QueueProxy[A] extends scala.collection.mutable.Queue[A] with SeqProxy[A] with ScalaObjectQueueProxy[>: Nothing <: AnyA] extends scala.collection.mutable.Queue[A]Queue[A] with SeqProxy[A]SeqProxy[A] {

  def => scala.collection.mutable.Queue[A]self: scala.collection.mutable.Queue[A]Queue[A]

  /** Access element number <code>n</code>.
   *
   *  @return  the element at index <code>n</code>.
   */
  override def (Int)Aapply(Intn: IntInt): AA = => scala.collection.mutable.Queue[A]self.(Int)Aapply(Intn)

  /** Returns the length of this queue.
   */
  override def => Intlength: IntInt = => scala.collection.mutable.Queue[A]self.=> Intlength

  /** Checks if the queue is empty.
   *
   *  @return true, iff there is no element in the queue.
   */
  override def => BooleanisEmpty: BooleanBoolean = => scala.collection.mutable.Queue[A]self.=> BooleanisEmpty

  /** Inserts a single element at the end of the queue.
   *
   *  @param  elem        the element to insert
   */
  override def (A)Unit+=(Aelem: AA): UnitUnit = => scala.collection.mutable.Queue[A]self (A)Unit+= Aelem

  /** Adds all elements provided by an <code>Iterable</code> object
   *  at the end of the queue. The elements are prepended in the order they
   *  are given out by the iterator.
   *
   *  @param  iter        an iterable object
   */
  override def (Iterable[A])Unit++=(Iterable[A]iter: Iterable[A]Iterable[A]): UnitUnit = => scala.collection.mutable.Queue[A]self (Iterable[A])Unit++= Iterable[A]iter

  /** Adds all elements provided by an iterator
   *  at the end of the queue. The elements are prepended in the order they
   *  are given out by the iterator.
   *
   *  @param  iter        an iterator
   */
  override def (Iterator[A])Unit++=(Iterator[A]it: Iterator[A]Iterator[A]): UnitUnit = => scala.collection.mutable.Queue[A]self (Iterator[A])Unit++= Iterator[A]it

  /** Adds all elements to the queue.
   *
   *  @param  elems       the elements to add.
   */
  override def (A*)Unitenqueue(A*elems: A*A*): UnitUnit = => scala.collection.mutable.Queue[A]self (Iterable[A])Unit++= A*elems

  /** Returns the first element in the queue, and removes this element
   *  from the queue.
   *
   *  @return the first element of the queue.
   */
  override def ()Adequeue(): AA = => scala.collection.mutable.Queue[A]self.()Adequeue

  /** Returns the first element in the queue, or throws an error if there
   *  is no element contained in the queue.
   *
   *  @return the first element.
   */
  override def => Afront: AA = => scala.collection.mutable.Queue[A]self.=> Afront

  /** Removes all elements from the queue. After this operation is completed,
   *  the queue will be empty.
   */
  override def ()Unitclear(): UnitUnit = => scala.collection.mutable.Queue[A]self.()Unitclear

  /** Returns an iterator over all elements on the queue.
   *
   *  @return an iterator over all queue elements.
   */
  override def => Iterator[A]elements: Iterator[A]Iterator[A] = => scala.collection.mutable.Queue[A]self.=> Iterator[A]elements

  /** This method clones the queue.
   *
   *  @return  a queue with the same elements.
   */
  override def ()scala.collection.mutable.Queue[A]clone(): scala.collection.mutable.Queue[A]Queue[A] = template $anon extends scala.collection.mutable.Queue[A] with scala.collection.mutable.QueueProxy[A]new scala.collection.mutable.QueueProxy[A]QueueProxy[A] {
    def => scala.collection.mutable.Queue[A]self = QueueProxy.this.typeQueueProxy.this.=> scala.collection.mutable.Queue[A]self.()scala.collection.mutable.Queue[A]clone()
  }
}
