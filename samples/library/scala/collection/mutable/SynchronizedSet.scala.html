package scala.collection.mutable


/** This class should be used as a mixin. It synchronizes the <code>Set</code>
 *  functions of the class into which it is mixed in.
 *
 *  @author  Matthias Zenger
 *  @version 1.0, 08/07/2003
 */
trait trait SynchronizedSet[A] extends java.lang.Object with scala.collection.mutable.Set[A] with ScalaObjectSynchronizedSet[>: Nothing <: AnyA] extends scala.collection.mutable.Set[A]Set[A] {

  abstract override def => Intsize: IntInt = (Int)Intsynchronized {
    => Intsuper.=> Intsize
  }

  override def => BooleanisEmpty: BooleanBoolean = (Boolean)Booleansynchronized {
    => Booleansuper.=> BooleanisEmpty
  }

  abstract override def (A)Booleancontains(Aelem: AA) = (Boolean)Booleansynchronized {
    (A)Booleansuper.(A)Booleancontains(Aelem)
  }

  abstract override def (A,Boolean)Unitupdate(Aelem: AA, Booleanincluded: BooleanBoolean): UnitUnit = (Unit)Unitsynchronized {
    (A,Boolean)Unitsuper.(A,Boolean)Unitupdate(Aelem, Booleanincluded)
  }

  abstract override def (A)Unit+=(Aelem: AA): UnitUnit = (Unit)Unitsynchronized {
    (A)Unitsuper.(A)Unit+=(Aelem)
  }

  override def (Iterable[A])Unit++=(Iterable[A]that: Iterable[A]Iterable[A]) = (Unit)Unitsynchronized {
    (Iterable[A])Unitsuper.(Iterable[A])Unit++=(Iterable[A]that)
  }

  override def (Iterator[A])Unit++=(Iterator[A]it: Iterator[A]Iterator[A]) = (Unit)Unitsynchronized {
    (Iterator[A])Unitsuper.(Iterator[A])Unit++=(Iterator[A]it)
  }

  override def (A*)Unitincl(A*elems: A*A*): UnitUnit = (Unit)Unitsynchronized {
    (Iterable[A])Unitsuper.++=(A*elems)
  }

  abstract override def (A)Unit-=(Aelem: AA): UnitUnit = (Unit)Unitsynchronized {
    (A)Unitsuper.(A)Unit-=(Aelem)
  }

  override def (Iterable[A])Unit--=(Iterable[A]that: Iterable[A]Iterable[A]) = (Unit)Unitsynchronized {
    (Iterable[A])Unitsuper.(Iterable[A])Unit--=(Iterable[A]that)
  }

  override def (Iterator[A])Unit--=(Iterator[A]it: Iterator[A]Iterator[A]) = (Unit)Unitsynchronized {
    (Iterator[A])Unitsuper.(Iterator[A])Unit--=(Iterator[A]it)
  }

  override def (A*)Unitexcl(A*elems: A*A*): UnitUnit = (Unit)Unitsynchronized {
    (Iterable[A])Unitsuper.--=(A*elems)
  }

  override def (scala.collection.mutable.Set[A])Unitintersect(scala.collection.mutable.Set[A]that: scala.collection.mutable.Set[A]Set[A]) = (Unit)Unitsynchronized {
    (scala.collection.mutable.Set[A])Unitsuper.(scala.collection.mutable.Set[A])Unitintersect(scala.collection.mutable.Set[A]that)
  }

  abstract override def ()Unitclear(): UnitUnit = (Unit)Unitsynchronized {
    ()Unitsuper.()Unitclear
  }

  override def (scala.collection.Set[A])BooleansubsetOf(scala.collection.Set[A]that: scala.collection.scala.collection.Set[A]Set[A]) = (Boolean)Booleansynchronized {
    (scala.collection.Set[A])Booleansuper.(scala.collection.Set[A])BooleansubsetOf(scala.collection.Set[A]that)
  }

  override def ((A) => Unit)Unitforeach((A) => Unitf: A => Unit) = (Unit)Unitsynchronized {
    ((A) => Unit)Unitsuper.((A) => Unit)Unitforeach((A) => Unitf)
  }

  override def ((A) => Boolean)Unitretain((A) => Booleanp: A => Boolean) = (Unit)Unitsynchronized {
    ((A) => Boolean)Unitsuper.((A) => Boolean)Unitretain((A) => Booleanp)
  }

  override def => List[A]toList: List[A]List[A] = (List[A])List[A]synchronized {
    => List[A]super.=> List[A]toList
  }

  override def ()StringtoString = (String)Stringsynchronized {
    ()Stringsuper.()StringtoString
  }

  override def (scala.collection.mutable.Message[A])Unit<<(scala.collection.mutable.Message[A]cmd: scala.collection.mutable.Message[A]Message[A]): UnitUnit = (Unit)Unitsynchronized {
    (scala.collection.mutable.Message[A])Unitsuper.(scala.collection.mutable.Message[A])Unit<<(scala.collection.mutable.Message[A]cmd)
  }

  override def ()scala.collection.mutable.Set[A]clone(): scala.collection.mutable.Set[A]Set[A] = (scala.collection.mutable.Set[A])scala.collection.mutable.Set[A]synchronized {
    ()scala.collection.mutable.Set[A]super.()scala.collection.mutable.Set[A]clone()
  }
}
