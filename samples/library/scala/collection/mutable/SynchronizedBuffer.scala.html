package scala.collection.mutable


/** This class should be used as a mixin. It synchronizes the <code>Buffer</code>
 *  methods of the class into which it is mixed in.
 *
 *  @author  Matthias Zenger
 *  @version 1.0, 08/07/2003
 */
trait trait SynchronizedBuffer[A] extends java.lang.Object with scala.collection.mutable.Buffer[A] with ScalaObjectSynchronizedBuffer[>: Nothing <: AnyA] extends scala.collection.mutable.Buffer[A]Buffer[A] {

  abstract override def => Intlength: IntInt = (Int)Intsynchronized {
    => Intsuper.=> Intlength
  }

  abstract override def => Iterator[A]elements: Iterator[A]Iterator[A] = (Iterator[A])Iterator[A]synchronized {
    => Iterator[A]super.=> Iterator[A]elements
  }

  abstract override def (Int)Aapply(Intn: IntInt): AA = (A)Asynchronized {
    (Int)Asuper.(Int)Aapply(Intn)
  }

  /** Append a single element to this buffer and return
   *  the identity of the buffer.
   *
   *  @param elem  the element to append.
   */
  override def (A)scala.collection.mutable.Buffer[A]+(Aelem: AA): scala.collection.mutable.Buffer[A]Buffer[A] = (scala.collection.mutable.Buffer[A])scala.collection.mutable.Buffer[A]synchronized {
    (A)scala.collection.mutable.Buffer[A]super.(A)scala.collection.mutable.Buffer[A]+(Aelem)
  }

  /** Append a single element to this buffer.
   *
   *  @param elem  the element to append.
   */
  abstract override def (A)Unit+=(Aelem: AA): UnitUnit = (Unit)Unitsynchronized {
    (A)Unitsuper.(A)Unit+=(Aelem)
  }

  /** Appends a number of elements provided by an iterable object

   *  buffer is returned.
   *
   *  @param iter  the iterable object.
   */
  override def (Iterable[A])scala.collection.mutable.Buffer[A]++(Iterable[A]iter: Iterable[A]Iterable[A]): scala.collection.mutable.Buffer[A]Buffer[A] = (scala.collection.mutable.Buffer[A])scala.collection.mutable.Buffer[A]synchronized {
    (Iterable[A])scala.collection.mutable.Buffer[A]super.(Iterable[A])scala.collection.mutable.Buffer[A]++(Iterable[A]iter)
  }

  /** Appends a number of elements provided by an iterable object
   *  via its <code>elements</code> method.
   *
   *  @param iter  the iterable object.
   */
  override def (Iterable[A])Unit++=(Iterable[A]iter: Iterable[A]Iterable[A]): UnitUnit = (Unit)Unitsynchronized {
    (Iterable[A])Unitsuper.(Iterable[A])Unit++=(Iterable[A]iter)
  }

  /** Appends a sequence of elements to this buffer.
   *
   *  @param elems  the elements to append.
   */
  override def (A*)Unitappend(A*elems: A*A*): UnitUnit = (Unit)Unitsynchronized {
    (Iterable[A])Unitsuper.++=(A*elems)
  }

  /** Appends a number of elements provided by an iterable object
   *  via its <code>elements</code> method.
   *
   *  @param iter  the iterable object.
   */
  override def (Iterable[A])UnitappendAll(Iterable[A]iter: Iterable[A]Iterable[A]): UnitUnit = (Unit)Unitsynchronized {
    (Iterable[A])Unitsuper.(Iterable[A])UnitappendAll(Iterable[A]iter)
  }

  /** Prepend a single element to this buffer and return
   *  the identity of the buffer.
   *
   *  @param elem  the element to append.
   */
  abstract override def (A)scala.collection.mutable.Buffer[A]+:(Aelem: AA): scala.collection.mutable.Buffer[A]Buffer[A] = (scala.collection.mutable.Buffer[A])scala.collection.mutable.Buffer[A]synchronized {
    (A)scala.collection.mutable.Buffer[A]super.(A)scala.collection.mutable.Buffer[A]+:(Aelem)
  }

  /** Prepends a number of elements provided by an iterable object
   *  via its <code>elements</code> method. The identity of the
   *  buffer is returned.
   *
   *  @param iter  the iterable object.
   */
  override def (Iterable[A])scala.collection.mutable.Buffer[A]++:(Iterable[A]iter: Iterable[A]Iterable[A]): scala.collection.mutable.Buffer[A]Buffer[A] = (scala.collection.mutable.Buffer[A])scala.collection.mutable.Buffer[A]synchronized {
    (Iterable[A])scala.collection.mutable.Buffer[A]super.(Iterable[A])scala.collection.mutable.Buffer[A]++:(Iterable[A]iter)
  }

  /** Prepend an element to this list.
   *
   *  @param elem  the element to prepend.

  }

  /** Prepends a number of elements provided by an iterable object
   *  via its <code>elements</code> method. The identity of the

  /** Inserts new elements at the index <code>n</code>. Opposed to method

   *  @param n      the index where a new element will be inserted.

  /** Inserts new elements at the index <code>n</code>. Opposed to method
   *  <code>update</code>, this method will not replace an element with a
   *  one. Instead, it will insert a new element at index <code>n</code>.
   *
   *  @param n     the index where a new element will be inserted.
   *  @param iter  the iterable object providing all elements to insert.

  /** Removes the element on a given index position.

   *  @param n  the index which refers to the element to delete.
   */
  abstract override def (Int)Aremove(Intn: IntInt): AA = (A)Asynchronized {
    (Int)Asuper.(Int)Aremove(Intn)
  }

  /** Clears the buffer contents.
   */
  abstract override def ()Unitclear(): UnitUnit = (Unit)Unitsynchronized {
    ()Unitsuper.()Unitclear
  }

  override def (scala.collection.mutable.Message[(scala.collection.mutable.Location, A)])Unit<<(scala.collection.mutable.Message[(scala.collection.mutable.Location, A)]cmd: scala.collection.mutable.Message[(scala.collection.mutable.Location, A)]Message[(Location, A)]): UnitUnit = (Unit)Unitsynchronized {

  /** Return a clone of this buffer.

   */

  /** The hashCode method always yields an error, since it is not
   *  safe to use buffers as keys in hash tables.
   *
   *  @return never.
   */
  override def ()InthashCode(): IntInt = (Int)Intsynchronized {
    ()Intsuper.()InthashCode()
  }
}
