package scala.collection.mutable


/** This class implements priority queues using a heap. The
 *  elements of the queue have to be ordered in terms of the
 *  <code>Ordered[T]</code> class.
 *
 *  @author  Matthias Zenger
 *  @version 1.0, 03/05/2004
 */
abstract class class PriorityQueueProxy[A] extends scala.collection.mutable.PriorityQueue[A] with RandomAccessSeqProxy[A] with ScalaObjectPriorityQueueProxy[>: Nothing <: AnyA <% Ordered[A]] extends scala.collection.mutable.PriorityQueue[A]PriorityQueue[A]
         with RandomAccessSeqProxy[A]RandomAccessSeqProxy[A]
{

  def => scala.collection.mutable.PriorityQueue[A]self: scala.collection.mutable.PriorityQueue[A]PriorityQueue[A]

  /** Creates a new iterator over all elements contained in this
   *  object.
   *
   *  @return the new iterator
   */
  override def => Iterator[A]elements: Iterator[A]Iterator[A] = => scala.collection.mutable.PriorityQueue[A]self.=> Iterator[A]elements

  /** Returns the length of this priority queue.
   */
  override def => Intlength: IntInt = => scala.collection.mutable.PriorityQueue[A]self.=> Intlength

  /** Checks if the queue is empty.
   *
   *  @return true, iff there is no element in the queue.
   */
  override def => BooleanisEmpty: BooleanBoolean = => scala.collection.mutable.PriorityQueue[A]self.=> BooleanisEmpty

  /** Inserts a single element into the priority queue.
   *
   *  @param  elem        the element to insert
   */
  override def (A)Unit+=(Aelem: AA): UnitUnit = => scala.collection.mutable.PriorityQueue[A]self (A)Unit+= Aelem

  /** Adds all elements provided by an <code>Iterable</code> object
   *  into the priority queue.
   *
   *  @param  iter        an iterable object
   */
  override def (Iterable[A])Unit++=(Iterable[A]iter: Iterable[A]Iterable[A]): UnitUnit = => scala.collection.mutable.PriorityQueue[A]self (Iterable[A])Unit++= Iterable[A]iter

  /** Adds all elements provided by an iterator into the priority queue.
   *
   *  @param  it        an iterator
   */
  override def (Iterator[A])Unit++=(Iterator[A]it: Iterator[A]Iterator[A]): UnitUnit = => scala.collection.mutable.PriorityQueue[A]self (Iterator[A])Unit++= Iterator[A]it

  /** Adds all elements to the queue.
   *
   *  @param  elems       the elements to add.
   */
  override def (A*)Unitenqueue(A*elems: A*A*): UnitUnit = => scala.collection.mutable.PriorityQueue[A]self (Iterable[A])Unit++= A*elems

  /** Returns the element with the highest priority in the queue,
   *  and removes this element from the queue.
   *
   *  @return   the element with the highest priority.
   */
  override def ()Adequeue(): AA = => scala.collection.mutable.PriorityQueue[A]self.()Adequeue

  /** Returns the element with the highest priority in the queue,
   *  or throws an error if there is no element contained in the queue.
   *
   *  @return   the element with the highest priority.
   */
  override def => Amax: AA = => scala.collection.mutable.PriorityQueue[A]self.=> Amax

  /** Removes all elements from the queue. After this operation is completed,
   *  the queue will be empty.
   */
  override def ()Unitclear(): UnitUnit = => scala.collection.mutable.PriorityQueue[A]self.()Unitclear

  /** Returns a regular queue containing the same elements.
   */
  override def => scala.collection.mutable.Queue[A]toQueue: scala.collection.mutable.Queue[A]Queue[A] = => scala.collection.mutable.PriorityQueue[A]self.=> scala.collection.mutable.Queue[A]toQueue

  /** This method clones the priority queue.

   *  @return  a priority queue with the same elements.
   */
  override def ()scala.collection.mutable.PriorityQueue[A]clone(): scala.collection.mutable.PriorityQueue[A]PriorityQueue[A] = template $anon extends scala.collection.mutable.PriorityQueueProxy[A]new scala.collection.mutable.PriorityQueueProxy[A]PriorityQueueProxy[A] {
