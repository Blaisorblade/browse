// $Id: SynchronizedPriorityQueue.scala 16893 2009-01-13 13:09:22Z cunei $


package scala.collection.mutable


/** This class implements synchronized priority queues using a heap.
 *  The elements of the queue have to be ordered in terms of the
 *  <code>Ordered[T]</code> class.
 *
 *  @author  Matthias Zenger
 *  @version 1.0, 03/05/2004
 */
class class SynchronizedPriorityQueue[A] extends scala.collection.mutable.PriorityQueue[A] with ScalaObjectSynchronizedPriorityQueue[>: Nothing <: AnyA <% Ordered[A]] extends scala.collection.mutable.PriorityQueue[A]PriorityQueue[A] {

  /** Checks if the queue is empty.
   *
   *  @return true, iff there is no element in the queue.
   */
  override def => BooleanisEmpty: BooleanBoolean = (Boolean)Booleansynchronized { super.=> BooleanisEmpty }

  /** Inserts a single element into the priority queue.
   *
   *  @param  elem        the element to insert
   */
  override def (A)Unit+=(Aelem: AA): UnitUnit = (Unit)Unitsynchronized { super.(A)Unit+=(Aelem) }

  /** Adds all elements provided by an <code>Iterable</code> object
   *  into the priority queue.
   *
   *  @param  iter        an iterable object
   */
  override def (Iterable[A])Unit++=(Iterable[A]iter: Iterable[A]Iterable[A]): UnitUnit = (Unit)Unitsynchronized { super.(Iterable[A])Unit++=(Iterable[A]iter) }

  /** Adds all elements provided by an iterator into the priority queue.
   *
   *  @param  it        an iterator
   */
  override def (Iterator[A])Unit++=(Iterator[A]it: Iterator[A]Iterator[A]): UnitUnit = (Unit)Unitsynchronized { super.(Iterator[A])Unit++=(Iterator[A]it) }

  /** Adds all elements to the queue.
   *
   *  @param  elems       the elements to add.
   */
  override def (A*)Unitenqueue(A*elems: A*A*): UnitUnit = (Unit)Unitsynchronized { super.(Iterable[A])Unit++=(A*elems) }

  /** Returns the element with the highest priority in the queue,
   *  and removes this element from the queue.
   *
   *  @return   the element with the highest priority.
   */
  override def ()Adequeue(): AA = (A)Asynchronized { super.()Adequeue }

  /** Returns the element with the highest priority in the queue,
   *  or throws an error if there is no element contained in the queue.
   *
   *  @return   the element with the highest priority.
   */
  override def => Amax: AA = (A)Asynchronized { super.=> Amax }

  /** Removes all elements from the queue. After this operation is completed,
   *  the queue will be empty.
   */
  override def ()Unitclear(): UnitUnit = (Unit)Unitsynchronized { super.()Unitclear }

  /** Returns an iterator which yiels all the elements of the priority
   *  queue in descending priority order.
   *
   *  @return  an iterator over all elements sorted in descending order.
   */
  override def => Iterator[A]elements: Iterator[A]Iterator[A] = (Iterator[A])Iterator[A]synchronized { super.=> Iterator[A]elements }

  /** Checks if two queues are structurally identical.
   *
   *  @return true, iff both queues contain the same sequence of elements.
   */
  override def (Any)Booleanequals(Anythat: AnyAny): BooleanBoolean = (Boolean)Booleansynchronized { super.(Any)Booleanequals(Anythat) }

  /** The hashCode method always yields an error, since it is not
   *  safe to use mutable queues as keys in hash tables.
   *
   *  @return never.
   */
  override def ()InthashCode(): IntInt = (Int)Intsynchronized { super.()InthashCode() }

  /** Returns a textual representation of a queue as a string.
   *
   *  @return the string representation of this queue.
   */
  override def ()StringtoString(): StringString = (String)Stringsynchronized { super.()StringtoString() }
}
