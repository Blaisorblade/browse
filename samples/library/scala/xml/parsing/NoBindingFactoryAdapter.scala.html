package scala.xml.parsing;


import scala.xml.factory.NodeFactory;
import org.xml.sax.InputSource;

/** nobinding adaptor providing callbacks to parser to create elements.
*   implements hash-consing
*/
class class NoBindingFactoryAdapter extends scala.xml.parsing.FactoryAdapter with scala.xml.factory.NodeFactory[scala.xml.Elem] with ScalaObjectNoBindingFactoryAdapter extends scala.xml.parsing.FactoryAdapterFactoryAdapter with scala.xml.factory.NodeFactory[scala.xml.Elem]NodeFactory[Elem] {
  
  // -- FactoryAdapter methods

  /** returns true. Every XML node may contain text that the application needs
  **/
  def (java.lang.String)BooleannodeContainsText( java.lang.Stringlabel:java.lang.java.lang.StringString ): BooleanBoolean = Boolean(true)true;
  

  // methods for NodeFactory[Elem]

  /** constructs an instance of scala.xml.Elem */
  protected def (String,String,scala.xml.MetaData,scala.xml.NamespaceBinding,Seq[scala.xml.Node])scala.xml.Elemcreate(Stringpre: StringString, Stringlabel: StringString, scala.xml.MetaDataattrs: scala.xml.MetaDataMetaData, scala.xml.NamespaceBindingscpe: scala.xml.NamespaceBindingNamespaceBinding, Seq[scala.xml.Node]children:Seq[scala.xml.Node]Seq[Node]): scala.xml.ElemElem = {
     (String,String,scala.xml.MetaData,scala.xml.NamespaceBinding,scala.xml.Node*)scala.xml.ElemElem( Stringpre, Stringlabel, scala.xml.MetaDataattrs, scala.xml.NamespaceBindingscpe, Seq[scala.xml.Node]children:_* );
  }

  // -- methods for FactoryAdapter

  /** creates a node. never creates the same node twice, using hash-consing 
  */
  def (String,String,scala.xml.MetaData,scala.xml.NamespaceBinding,List[scala.xml.Node])scala.xml.ElemcreateNode(Stringpre:StringString, Stringlabel: StringString, scala.xml.MetaDataattrs: scala.xml.MetaDataMetaData, scala.xml.NamespaceBindingscpe: scala.xml.NamespaceBindingNamespaceBinding, List[scala.xml.Node]children: List[scala.xml.Node]List[Node] ): scala.xml.ElemElem = {
    //Console.println("NoBindingFactoryAdapter::createNode("+pre+","+label+","+attrs+","+scpe+","+children+")");
    (String,String,scala.xml.MetaData,scala.xml.NamespaceBinding,scala.xml.Node*)scala.xml.ElemElem( Stringpre, Stringlabel, scala.xml.MetaDataattrs, scala.xml.NamespaceBindingscpe, List[scala.xml.Node]children:_* );
    //makeNode(pre, label, attrs, scpe, children);
  }
  
  /** creates a text node
  */
  def (String)scala.xml.TextcreateText( Stringtext:StringString ) = 
    (String)scala.xml.TextText( Stringtext );

  /** create a processing instruction
  */
  def (String,String)Seq[scala.xml.ProcInstr]createProcInstr(Stringtarget: StringString, Stringdata: StringString) = 
    (String,String)Seq[scala.xml.ProcInstr]makeProcInstr(Stringtarget, Stringdata)
    
  /** loads an XML document, returning a Symbol node.
  */
  override def (org.xml.sax.InputSource)scala.xml.ElemloadXML( org.xml.sax.InputSourcesource:org.xml.sax.InputSourceInputSource ):scala.xml.ElemElem = 
    super.(org.xml.sax.InputSource)scala.xml.NodeloadXML( org.xml.sax.InputSourcesource ).scala.xml.ElemasInstanceOf[ scala.xml.ElemElem ];

}
