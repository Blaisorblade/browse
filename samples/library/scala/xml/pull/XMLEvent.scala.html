package scala.xml.pull

/** This class represents an XML event for pull parsing.
 *  Pull parsing means that during the traversal of the XML
 *  tree we are parsing, each "event" is returned to the caller
 *  and the traversal is suspended.
 */
trait trait XMLEvent extends java.lang.ObjectXMLEvent

/** An element is encountered the first time */
case class class EvElemStart extends java.lang.Object with scala.xml.pull.XMLEvent with ScalaObject with ProductEvElemStart(Stringpre: StringString, Stringlabel: StringString, scala.xml.MetaDataattrs: scala.xml.MetaDataMetaData, scala.xml.NamespaceBindingscope: scala.xml.NamespaceBindingNamespaceBinding) extends scala.xml.pull.XMLEventXMLEvent 

/** An element is encountered the last time */
case class class EvElemEnd extends java.lang.Object with scala.xml.pull.XMLEvent with ScalaObject with ProductEvElemEnd(Stringpre: StringString, Stringlabel: StringString) extends scala.xml.pull.XMLEventXMLEvent 
/** A text node is encountered */
case class class EvText extends java.lang.Object with scala.xml.pull.XMLEvent with ScalaObject with ProductEvText(Stringtext: StringString) extends scala.xml.pull.XMLEventXMLEvent

/** An entity reference is encountered */
case class class EvEntityRef extends java.lang.Object with scala.xml.pull.XMLEvent with ScalaObject with ProductEvEntityRef(Stringentity: StringString) extends scala.xml.pull.XMLEventXMLEvent

/** A processing instruction is encountered */
case class class EvProcInstr extends java.lang.Object with scala.xml.pull.XMLEvent with ScalaObject with ProductEvProcInstr(Stringtarget: StringString, Stringtext: StringString) extends scala.xml.pull.XMLEventXMLEvent

/** A comment is encountered */
case class class EvComment extends java.lang.Object with scala.xml.pull.XMLEvent with ScalaObject with ProductEvComment(Stringtext: StringString) extends scala.xml.pull.XMLEventXMLEvent
