package scala.xml.transform

/** a RewriteRule, when applied to a term, yields either
 *  the resulting of rewriting or the term itself it the rule
 *  is not applied.
 *
 *  @author  Burak Emir
 *  @version 1.0
 */
abstract class class RewriteRule extends scala.xml.transform.BasicTransformer with ScalaObjectRewriteRule extends scala.xml.transform.BasicTransformerBasicTransformer {
  /** a name for this rewrite rule */
  val java.lang.Stringname = RewriteRule.this.typethis.()java.lang.StringtoString()
  override def (Seq[scala.xml.Node])Seq[scala.xml.Node]transform(Seq[scala.xml.Node]ns: Seq[scala.xml.Node]Seq[Node]): Seq[scala.xml.Node]Seq[Node] = super.(Seq[scala.xml.Node])Seq[scala.xml.Node]transform(Seq[scala.xml.Node]ns)
  override def (scala.xml.Node)Seq[scala.xml.Node]transform(scala.xml.Noden: scala.xml.NodeNode): Seq[scala.xml.Node]Seq[Node] = scala.xml.Noden
}
