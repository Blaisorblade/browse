package scala.xml

/** <p>
 *    <code>SpecialNode</code> is a special XML node which
 *    represents either text (PCDATA), a comment, a PI, or an entity ref.
 *  </p>
 *  <p>
 *    SpecialNodes also play the role of XMLEvents for pull-parsing.
 *  </p>
 *
 *  @author Burak Emir
 */
abstract class class SpecialNode extends scala.xml.Node with scala.xml.pull.XMLEvent with ScalaObjectSpecialNode extends scala.xml.NodeNode with pull.scala.xml.pull.XMLEventXMLEvent {

  /** always empty */
  final override def => object scala.xml.Nullattributes = object scala.xml.NullNull

  /** always Node.EmptyNamespace */
  final override def => Nullnamespace = Null(null)null

  /** always empty */
  final def => object Nilchild = object NilNil

  /** append string representation to the given stringbuffer */
  def (StringBuilder)StringBuildertoString(StringBuildersb: StringBuilderStringBuilder): StringBuilderStringBuilder

}
