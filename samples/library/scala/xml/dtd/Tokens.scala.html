package scala.xml.dtd


class class Tokens extends java.lang.Object with ScalaObjectTokens {

  // Tokens

  final val Int(0)TOKEN_PCDATA = Int(0)0
  final val Int(1)NAME         = Int(1)1
  final val Int(3)LPAREN       = Int(3)3
  final val Int(4)RPAREN       = Int(4)4
  final val Int(5)COMMA        = Int(5)5
  final val Int(6)STAR         = Int(6)6
  final val Int(7)PLUS         = Int(7)7
  final val Int(8)OPT          = Int(8)8
  final val Int(9)CHOICE       = Int(9)9
  final val Int(10)END          = Int(10)10
  final val Int(13)S            = Int(13)13

  final def (Int)Stringtoken2string(Inti: IntInt): StringString = Inti Stringmatch {
    java.lang.String("#PCDATA")case  Int(0)0 => java.lang.String("#PCDATA")"#PCDATA"
    java.lang.String("NAME")case  Int(1)1 => java.lang.String("NAME")"NAME"
    java.lang.String("(")case  Int(3)3 => java.lang.String("(")"("
    java.lang.String(")")case  Int(4)4 => java.lang.String(")")")"
    java.lang.String(",")case  Int(5)5 => java.lang.String(",")","
    java.lang.String("*")case  Int(6)6 => java.lang.String("*")"*"
    java.lang.String("+")case  Int(7)7 => java.lang.String("+")"+"
    java.lang.String("?")case  Int(8)8 => java.lang.String("?")"?"
    java.lang.String("|")case  Int(9)9 => java.lang.String("|")"|"
    java.lang.String("END")case Int(10)10 => java.lang.String("END")"END"
    java.lang.String(" ")case Int(13)13 => java.lang.String(" ")" "
  }
}
