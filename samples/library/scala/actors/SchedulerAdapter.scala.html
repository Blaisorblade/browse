package scala.actors

/** The <code>SchedulerAdapter</code> trait is used to adapt
 *  the behavior of the standard <code>Scheduler</code> object.
 *
 *  Providing an implementation for the
 *  <code>execute(f: => Unit)</code> method is sufficient to
 *  obtain a concrete class that extends <code>IScheduler</code>.
 *
 *  @version 0.9.18
 *  @author Philipp Haller
 */
trait trait SchedulerAdapter extends java.lang.Object with scala.actors.IScheduler with ScalaObjectSchedulerAdapter extends scala.actors.ISchedulerIScheduler {

  /** Submits a <code>Runnable</code> for execution.
   *
   *  @param  task  the task to be executed
   */
  def (Runnable)Unitexecute(Runnabletask: RunnableRunnable): UnitUnit =
    (=> Unit)Unitexecute { Runnabletask.()Unitrun() }

  /** Shuts down the scheduler.
   */
  def ()Unitshutdown(): UnitUnit =
    object scala.actors.SchedulerScheduler.()Unitshutdown()

  /** The <code>ActorGC</code> instance that keeps track of the
   *  live actor objects that are managed by <code>this</code>
   *  scheduler.
   */
  val scala.actors.ActorGCactorGC: scala.actors.ActorGCActorGC = scala.actors.ActorGCnew scala.actors.ActorGCActorGC

  def (() => Unit)UnitonLockup(() => Unithandler: () => Unit) Unit{}

  def (Int)(() => Unit)UnitonLockup(Intmillis: IntInt)(() => Unithandler: () => Unit) Unit{}

  def => UnitprintActorDump Unit{}

}
