package scala.reflect

import Predef._

/** This type is required by the compiler and <b>should not be used in client code</b>. */
abstract class class Type extends java.lang.Object with ScalaObjectType

/** This type is required by the compiler and <b>should not be used in client code</b>. */
case object object scala.reflect.NoPrefixNoPrefix extends scala.reflect.TypeType
/** This type is required by the compiler and <b>should not be used in client code</b>. */
case object object scala.reflect.NoTypeNoType extends scala.reflect.TypeType

/** This type is required by the compiler and <b>should not be used in client code</b>.
  * fullname */
case class class NamedType extends scala.reflect.Type with ScalaObject with ProductNamedType(Stringfullname: StringString) extends scala.reflect.TypeType

/** This type is required by the compiler and <b>should not be used in client code</b>.
  * pre # sym */
case class class PrefixedType extends scala.reflect.Type with ScalaObject with ProductPrefixedType(scala.reflect.Typepre: scala.reflect.TypeType, scala.reflect.Symbolsym: scala.reflect.SymbolSymbol) extends scala.reflect.TypeType

/** This type is required by the compiler and <b>should not be used in client code</b>.
  * pre.type # sym == pre.sym */
case class class SingleType extends scala.reflect.Type with ScalaObject with ProductSingleType(scala.reflect.Typepre: scala.reflect.TypeType, scala.reflect.Symbolsym: scala.reflect.SymbolSymbol) extends scala.reflect.TypeType

/** This type is required by the compiler and <b>should not be used in client code</b>.
  * clazz.this */
case class class ThisType extends scala.reflect.Type with ScalaObject with ProductThisType(scala.reflect.Symbolclazz: scala.reflect.SymbolSymbol) extends scala.reflect.TypeType

/** This type is required by the compiler and <b>should not be used in client code</b>.
  * clazz.super[superClazz]
  * <code>tpe[args1, ..., argsn]</code> */
case class class AppliedType extends scala.reflect.Type with ScalaObject with ProductAppliedType(scala.reflect.Typetpe: scala.reflect.TypeType, List[scala.reflect.Type]args: List[scala.reflect.Type]List[Type]) extends scala.reflect.TypeType

/** This type is required by the compiler and <b>should not be used in client code</b>.
  * [a &lt;: lo &gt;: hi] */
case class class TypeBounds extends scala.reflect.Type with ScalaObject with ProductTypeBounds(scala.reflect.Typelo: scala.reflect.TypeType, scala.reflect.Typehi: scala.reflect.TypeType) extends scala.reflect.TypeType

/** This type is required by the compiler and <b>should not be used in client code</b>.
  * <code>(formals1 ... formalsn) restpe</code> */
case class class MethodType extends scala.reflect.Type with ScalaObject with ProductMethodType(List[scala.reflect.Type]formals: List[scala.reflect.Type]List[Type], scala.reflect.Typerestpe: scala.reflect.TypeType) extends scala.reflect.TypeType


/** This type is required by the compiler and <b>should not be used in client code</b>. */
case class class PolyType extends scala.reflect.Type with ScalaObject with ProductPolyType(List[scala.reflect.Symbol]typeParams: List[scala.reflect.Symbol]List[Symbol], List[(scala.reflect.Type, scala.reflect.Type)]typeBounds: List[(scala.reflect.Type, scala.reflect.Type)]List[(Type, Type)], scala.reflect.TyperesultType: scala.reflect.TypeType) extends scala.reflect.TypeType


/** This type is required by the compiler and <b>should not be used in client code</b>. */
class class ImplicitMethodType extends scala.reflect.MethodType with ScalaObjectImplicitMethodType(List[scala.reflect.Type]formals: List[scala.reflect.Type]List[Type], scala.reflect.Typerestpe: scala.reflect.TypeType)
extends scala.reflect.MethodTypeMethodType(List[scala.reflect.Type]formals, scala.reflect.Typerestpe)

  case reflect.NoType =>

  case reflect.ThisType(clazz) => 
  case reflect.AppliedType(tpe, args) => 
  case reflect.TypeBounds(lo, hi) => 
  case reflect.MethodType(formals, restpe) => //can also be ImplicitMethodType
  case reflect.PolyType(typeParams, typeBounds, resultType) =>
