package scala.reflect

/** Provides a short description that will be included when generating
 *  bean information. This annotation can be attached to the bean itself,
 *  or to any member. 
 *
 *  @author Ross Judson (rjudson@managedobjects.com)
 */
class class BeanDescription extends Annotation with ScalaObjectBeanDescription(val Stringdescription: StringString) extends AnnotationAnnotation
