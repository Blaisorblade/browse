package scala.util.parsing.syntax

/** This component provides the notion of `Token', the unit of information that is passed from lexical
 * parsers in the `Lexical' component to the parsers in the `TokenParsers' component.
 *
 * @author Martin Odersky, Adriaan Moors
 */
trait trait Tokens extends java.lang.Object with ScalaObjectTokens {
  /** Objects of this type are produced by a lexical parser or ``scanner'', and consumed by a parser
   * {@see scala.util.parsing.combinator.syntactical.TokenParsers}.
   */
  abstract class class Token extends java.lang.Object with ScalaObjectToken {
    def => Stringchars: StringString
  }
  
  /** A class of error tokens. Error tokens are used to communicate
   *  errors detected during lexical analysis 
   */
  case class class ErrorToken extends Tokens.this.Token with ScalaObject with ProductErrorToken(Stringmsg: StringString) extends Tokens.this.TokenToken {
    def => java.lang.Stringchars = java.lang.String("*** error: ")"*** error: "(Any)java.lang.String+=> Stringmsg
  }

  /** A class for end-of-file tokens */
  case object object Tokens.this.EOFEOF extends Tokens.this.TokenToken {
    def => java.lang.Stringchars = java.lang.String("<eof>")"<eof>"
  }

  /** This token is produced by a scanner {@see Scanner} when scanning failed. */
  def (String)Tokens.this.TokenerrorToken(Stringmsg: StringString): Tokens.this.TokenToken = Tokens.this.ErrorTokennew Tokens.this.ErrorTokenErrorToken(Stringmsg)  
}
