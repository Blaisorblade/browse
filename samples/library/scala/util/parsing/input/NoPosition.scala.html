package scala.util.parsing.input

/** Undefined position 
 *
 * @author Martin Odersky, Adriaan Moors
 */
object NoPosition extends Position {
  def line = 0
  def column = 0
  override def toString = "<undefined position>"
  override def longString = toString
  def lineContents = ""
}
