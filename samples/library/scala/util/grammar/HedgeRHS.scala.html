package scala.util.grammar

abstract class class HedgeRHS extends java.lang.Object with ScalaObjectHedgeRHS

/** right hand side of a hedge production, deriving a single tree */
case class class ConsRHS extends scala.util.grammar.HedgeRHS with ScalaObject with ProductConsRHS(Inttnt: IntInt, Inthnt: IntInt) extends scala.util.grammar.HedgeRHSHedgeRHS

/** right hand side of a hedge production, deriving any hedge */
case object object scala.util.grammar.AnyHedgeRHSAnyHedgeRHS extends scala.util.grammar.HedgeRHSHedgeRHS

/** right hand side of a hedge production, deriving the empty hedge */
case object object scala.util.grammar.EmptyHedgeRHSEmptyHedgeRHS extends scala.util.grammar.HedgeRHSHedgeRHS
