package scala.util.regexp

/** <p>
 *    The class <code>WordExp</code> provides regular word expressions.
 *    Users have to instantiate type member <code>_regexpT &lt;: RegExp</code>
 *    (from class <code>Base</code>) and a type member
 *    <code>_labelT &lt;: Label</code>. Here is a short example:
 *  </p><pre>
 *  <b>import</b> scala.util.regexp._
 *  <b>import</b> scala.util.automata._
 *  <b>object</b> MyLang <b>extends</b> WordExp {
 *    <b>type</b> _regexpT = RegExp
 *    <b>type</b> _labelT = MyChar
 *
 *    <b>case class</b> MyChar(c:Char) <b>extends</b> Label
 *  }
 *  <b>import</b> MyLang._
 *  // (a* | b)*
 *  <b>val</b> rex = Star(Alt(Star(Letter(MyChar('a'))),Letter(MyChar('b'))))
 *  <b>object</b> MyBerriSethi <b>extends</b> WordBerrySethi {
 *    <b>override val</b> lang = MyLang
 *  }
 *  <b>val</b> nfa = MyBerriSethi.automatonFrom(Sequ(rex), 1)
 *  </pre>
 *
 *  @author  Burak Emir
 *  @version 1.0
 */
abstract class class WordExp extends scala.util.regexp.Base with ScalaObjectWordExp extends scala.util.regexp.BaseBase {

  abstract class class Label extends java.lang.Object with ScalaObjectLabel

  type >: Nothing <: WordExp.this.RegExp_regexpT <: RegExp
  type >: Nothing <: WordExp.this.Label_labelT <: Label

  case class class Letter extends WordExp.this.RegExp with ScalaObject with ProductLetter(WordExp.this._labelTa: WordExp.this._labelT_labelT) extends WordExp.this.RegExpRegExp {
    final lazy val BooleanisNullable = Boolean(false)false
    var Intpos = -Int(-1)1
  }

  case class class Wildcard extends WordExp.this.RegExp with ScalaObject with ProductWildcard() extends WordExp.this.RegExpRegExp {
    final lazy val BooleanisNullable = Boolean(false)false
    var Intpos = -Int(-1)1
  }
}
