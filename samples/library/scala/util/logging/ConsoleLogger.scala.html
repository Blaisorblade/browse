package scala.util.logging

/**
 *  The trait <code>ConsoleLogger</code> is mixed into a concrete class who
 *  has class <code>Logged</code> among its base classes.
 *
 *  @author  Burak Emir
 *  @version 1.0
 */
trait trait ConsoleLogger extends java.lang.Object with scala.util.logging.Logged with ScalaObjectConsoleLogger extends scala.util.logging.LoggedLogged {

  /** logs argument to Console using <code>Console.println</code>
   *
   *  @param msg ...
   */
  override def (String)Unitlog(Stringmsg: StringString): UnitUnit = object ConsoleConsole.(Any)Unitprintln(Stringmsg)
}
