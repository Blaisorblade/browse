package scala

/** A module defining utility methods for higher-order functional programming.
 *
 *  @author  Martin Odersky
 *  @version 1.0, 29/11/2006
 */
object object FunctionFunction {

  /** Given a sequence of functions <code>f<sub>1</sub></code>, ...,
   *  <code>f<sub>n</sub></code>, return the function <code>f<sub>1</sub>
   *  andThen ... andThen f<sub>n</sub></code>.
   *
   *  @param fs The given sequence of functions
   *  @return   ...
   */
  def [a](Seq[(a) => a])(a) => achain[>: Nothing <: Anya](Seq[(a) => a]fs: Seq[(a) => a]Seq[a => a]): a => a = { ax => (ax (a)((a, (a) => a) => a)a/: Seq[(a) => a]fs) ((ax, (a) => af) => (a)af(ax)) }

  /** The constant function */
  def [T, U](T)(U)Tconst[>: Nothing <: AnyT, >: Nothing <: AnyU](Tx: TT)(Uy: UU): TT = Tx

  /** Currying for functions of arity 2. This transforms a function
   *  of arity 2 into a a unary function returning another unary function.
   *
   *  @param f  ...
   *  @return   ...
   */
  def [a1, a2, b]((a1, a2) => b)(a1) => (a2) => bcurried[>: Nothing <: Anya1, >: Nothing <: Anya2, >: Nothing <: Anyb]((a1, a2) => bf: (a1, a2) => b): a1 => a2 => b = {
    a1x1 => a2x2 => (a1,a2)bf(a1x1, a2x2)
  }

  /** Currying for functions of arity 3.
   *
   *  @param f  ...
   *  @return   ...
   */
  def [a1, a2, a3, b]((a1, a2, a3) => b)(a1) => (a2) => (a3) => bcurried[>: Nothing <: Anya1, >: Nothing <: Anya2, >: Nothing <: Anya3, >: Nothing <: Anyb]((a1, a2, a3) => bf: (a1, a2, a3) => b): a1 => a2 => a3 => b = {
    a1x1 => a2x2 => a3x3 => (a1,a2,a3)bf(a1x1, a2x2, a3x3)
  }

  /** Currying for functions of arity 4.
   */
  def [a1, a2, a3, a4, b]((a1, a2, a3, a4) => b)(a1) => (a2) => (a3) => (a4) => bcurried[>: Nothing <: Anya1, >: Nothing <: Anya2, >: Nothing <: Anya3, >: Nothing <: Anya4, >: Nothing <: Anyb]((a1, a2, a3, a4) => bf: (a1, a2, a3, a4) => b): a1 => a2 => a3 => a4 => b = {
    a1x1 => a2x2 => a3x3 => a4x4 => (a1,a2,a3,a4)bf(a1x1, a2x2, a3x3, a4x4)
  }

  /** Currying for functions of arity 5.
   */
  def [a1, a2, a3, a4, a5, b]((a1, a2, a3, a4, a5) => b)(a1) => (a2) => (a3) => (a4) => (a5) => bcurried[>: Nothing <: Anya1, >: Nothing <: Anya2, >: Nothing <: Anya3, >: Nothing <: Anya4, >: Nothing <: Anya5, >: Nothing <: Anyb]((a1, a2, a3, a4, a5) => bf: (a1, a2, a3, a4, a5) => b): a1 => a2 => a3 => a4 => a5 => b = {
    a1x1 => a2x2 => a3x3 => a4x4 => a5x5 => (a1,a2,a3,a4,a5)bf(a1x1, a2x2, a3x3, a4x4, a5x5)
  }

  /** Uncurrying for functions of arity 2. This transforms a unary function
   *  returning another unary function into a function of arity 2.
   */
  def [a1, a2, b]((a1) => (a2) => b)(a1, a2) => buncurried[>: Nothing <: Anya1, >: Nothing <: Anya2, >: Nothing <: Anyb]((a1) => (a2) => bf: a1 => a2 => b): (a1, a2) => b = {
    (a1x1, a2x2) => (a2)bf(a1x1)(a2x2)
  }

  /** Uncurrying for functions of arity 3.
   */
  def [a1, a2, a3, b]((a1) => (a2) => (a3) => b)(a1, a2, a3) => buncurried[>: Nothing <: Anya1, >: Nothing <: Anya2, >: Nothing <: Anya3, >: Nothing <: Anyb]((a1) => (a2) => (a3) => bf: a1 => a2 => a3 => b): (a1, a2, a3) => b = {
    (a1x1, a2x2, a3x3) => (a3)bf(a1x1)(a2x2)(a3x3)
  }

  /** Uncurrying for functions of arity 4.
   */
  def [a1, a2, a3, a4, b]((a1) => (a2) => (a3) => (a4) => b)(a1, a2, a3, a4) => buncurried[>: Nothing <: Anya1, >: Nothing <: Anya2, >: Nothing <: Anya3, >: Nothing <: Anya4, >: Nothing <: Anyb]((a1) => (a2) => (a3) => (a4) => bf: a1 => a2 => a3 => a4 => b): (a1, a2, a3, a4) => b = {
    (a1x1, a2x2, a3x3, a4x4) => (a4)bf(a1x1)(a2x2)(a3x3)(a4x4)
  }

  /** Uncurrying for functions of arity 5.
   */
  def [a1, a2, a3, a4, a5, b]((a1) => (a2) => (a3) => (a4) => (a5) => b)(a1, a2, a3, a4, a5) => buncurried[>: Nothing <: Anya1, >: Nothing <: Anya2, >: Nothing <: Anya3, >: Nothing <: Anya4, >: Nothing <: Anya5, >: Nothing <: Anyb]((a1) => (a2) => (a3) => (a4) => (a5) => bf: a1 => a2 => a3 => a4 => a5 => b): (a1, a2, a3, a4, a5) => b  =  {
    (a1x1, a2x2, a3x3, a4x4, a5x5) => (a5)bf(a1x1)(a2x2)(a3x3)(a4x4)(a5x5)
  }

  /** Tupling for functions of arity 2. This transforms a function
   *  of arity 2 into a unary function that takes a pair of arguments.
   *
   *  @param f  ...
   *  @return   ...
   */
  def [a1, a2, b]((a1, a2) => b)((a1, a2)) => btupled[>: Nothing <: Anya1, >: Nothing <: Anya2, >: Nothing <: Anyb]((a1, a2) => bf: (a1, a2) => b): Tuple2[a1, a2] => b = b{
    bcase Tuple2(a1x1, a2x2) => (a1,a2)bf(a1x1, a2x2)
  }

  /** Tupling for functions of arity 3. This transforms a function
   *  of arity 3 into a unary function that takes a triple of arguments.
   */
  def [a1, a2, a3, b]((a1, a2, a3) => b)((a1, a2, a3)) => btupled[>: Nothing <: Anya1, >: Nothing <: Anya2, >: Nothing <: Anya3, >: Nothing <: Anyb]((a1, a2, a3) => bf: (a1, a2, a3) => b): Tuple3[a1, a2, a3] => b = b{
    bcase Tuple3(a1x1, a2x2, a3x3) => (a1,a2,a3)bf(a1x1, a2x2, a3x3)
  }

  /** Tupling for functions of arity 4. This transforms a function
   *  of arity 4 into a unary function that takes a 4-tuple of arguments.
   */
  def [a1, a2, a3, a4, b]((a1, a2, a3, a4) => b)((a1, a2, a3, a4)) => btupled[>: Nothing <: Anya1, >: Nothing <: Anya2, >: Nothing <: Anya3, >: Nothing <: Anya4, >: Nothing <: Anyb]((a1, a2, a3, a4) => bf: (a1, a2, a3, a4) => b): Tuple4[a1, a2, a3, a4] => b = b{
    bcase Tuple4(a1x1, a2x2, a3x3, a4x4) => (a1,a2,a3,a4)bf(a1x1, a2x2, a3x3, a4x4)
  }

  /** Tupling for functions of arity 5. This transforms a function
   *  of arity 5 into a unary function that takes a 5-tuple of arguments.
   */
  def [a1, a2, a3, a4, a5, b]((a1, a2, a3, a4, a5) => b)((a1, a2, a3, a4, a5)) => btupled[>: Nothing <: Anya1, >: Nothing <: Anya2, >: Nothing <: Anya3, >: Nothing <: Anya4, >: Nothing <: Anya5, >: Nothing <: Anyb]((a1, a2, a3, a4, a5) => bf: (a1, a2, a3, a4, a5) => b): Tuple5[a1, a2, a3, a4, a5] => b = b{
    bcase Tuple5(a1x1, a2x2, a3x3, a4x4, a5x5) => (a1,a2,a3,a4,a5)bf(a1x1, a2x2, a3x3, a4x4, a5x5)
  }

  /** Un-tupling for functions of arity 2. This transforms a function taking
   *  a pair of arguments into a binary function which takes each argument separately.
   */
  def [a1, a2, b](((a1, a2)) => b)(a1, a2) => buntupled[>: Nothing <: Anya1, >: Nothing <: Anya2, >: Nothing <: Anyb](((a1, a2)) => bf: Tuple2[a1, a2] => b): (a1, a2) => b = {
    (a1x1, a2x2) => ((a1, a2))bf((a1,a2)(a1, a2)Tuple2(a1x1, a2x2))
  }

  /** Un-tupling for functions of arity 3. This transforms a function taking
   *  a triple of arguments into a ternary function which takes each argument separately.
   */
  def [a1, a2, a3, b](((a1, a2, a3)) => b)(a1, a2, a3) => buntupled[>: Nothing <: Anya1, >: Nothing <: Anya2, >: Nothing <: Anya3, >: Nothing <: Anyb](((a1, a2, a3)) => bf: Tuple3[a1, a2, a3] => b): (a1, a2, a3) => b = {
    (a1x1, a2x2, a3x3) => ((a1, a2, a3))bf((a1,a2,a3)(a1, a2, a3)Tuple3(a1x1, a2x2, a3x3))
  }

  /** Un-tupling for functions of arity 4. This transforms a function taking
   *  a 4-tuple of arguments into a function of arity 4 which takes each argument separately.
   */
  def [a1, a2, a3, a4, b](((a1, a2, a3, a4)) => b)(a1, a2, a3, a4) => buntupled[>: Nothing <: Anya1, >: Nothing <: Anya2, >: Nothing <: Anya3, >: Nothing <: Anya4, >: Nothing <: Anyb](((a1, a2, a3, a4)) => bf: Tuple4[a1, a2, a3, a4] => b): (a1, a2, a3, a4) => b = {
    (a1x1, a2x2, a3x3, a4x4) => ((a1, a2, a3, a4))bf((a1,a2,a3,a4)(a1, a2, a3, a4)Tuple4(a1x1, a2x2, a3x3, a4x4))
  }

  /** Un-tupling for functions of arity 5. This transforms a function taking
   *  a 5-tuple of arguments into a function of arity 5 which takes each argument separately.
   */
  def [a1, a2, a3, a4, a5, b](((a1, a2, a3, a4, a5)) => b)(a1, a2, a3, a4, a5) => buntupled[>: Nothing <: Anya1, >: Nothing <: Anya2, >: Nothing <: Anya3, >: Nothing <: Anya4, >: Nothing <: Anya5, >: Nothing <: Anyb](((a1, a2, a3, a4, a5)) => bf: Tuple5[a1, a2, a3, a4, a5] => b): (a1, a2, a3, a4, a5) => b = {
    (a1x1, a2x2, a3x3, a4x4, a5x5) => ((a1, a2, a3, a4, a5))bf((a1,a2,a3,a4,a5)(a1, a2, a3, a4, a5)Tuple5(a1x1, a2x2, a3x3, a4x4, a5x5))
  }
}
