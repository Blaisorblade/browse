package scala.runtime


final class class RichByte extends java.lang.Object with Proxy with Ordered[Byte] with ScalaObjectRichByte(Bytex: ByteByte) extends ProxyProxy with Ordered[Byte]Ordered[Byte] {

  // Proxy.self
  def => Anyself: AnyAny = Bytex

  // Ordered[Byte].compare
  def (Byte)Intcompare (Bytey: ByteByte): IntInt = Intif (Bytex (Byte)Boolean< Bytey) -Int(-1)1 else Intif (Bytex (Byte)Boolean> Bytey) Int(1)1 else Int(0)0

}
