package scala


import Predef._

/** This class implements a simple proxy that forwards all calls to
 *  methods of class <code>Any</code> to another object <code>self</code>.
 *  Please note that only those methods can be forwarded that are
 *  overridable and public.
 *
 *  @author  Matthias Zenger
 *  @version 1.0, 26/04/2004
 */
trait trait Proxy extends java.lang.Object with ScalaObjectProxy {
  def => Anyself: AnyAny
  override def ()InthashCode: IntInt = => Anyself.()InthashCode
  override def (Any)Booleanequals(Anythat: AnyAny): BooleanBoolean = Anythat Booleanmatch {
    Booleancase Proxythat: ProxyProxy => => Anyself (Any)Booleanequals Proxythat.=> Anyself
    Boolean(false)case Anythat        => Boolean(false)false
  }
  override def ()StringtoString: StringString = => Anyself.()java.lang.StringtoString
}
