/** This class implements a proxy for sequences. It forwards
 *  all calls to a different sequence object.
 *
 *  @author  Martin Odersky
 *  @author  Matthias Zenger
 *  @version 2.0, 31/12/2006
 */

  // Seq delegates
  // Seq methods could be printed by  cat Seq.scala | sed -n '/trait Seq/,$ p' | egrep '^  (override )?def'

  override def (Int)IntlengthCompare(Intl: IntInt) = => Seq[A]self (Int)IntlengthCompare Intl
  override def => Intsize = => Seq[A]self.=> Intsize
  override def => BooleanisEmpty = => Seq[A]self.=> BooleanisEmpty
  @deprecateddeprecated

  override def => Alast = => Seq[A]self.=> Alast
  override def => Option[A]lastOption = => Seq[A]self.=> Option[A]lastOption
